#version 430 core

layout (location = 3) uniform sampler2D albedo;
layout (location = 4) uniform vec3 lightDir;
layout (location = 5) uniform vec4 lightColor;

in vec2 vUV;
in vec3 vNormal;

out vec4 vertColor;

void main()
{
	float d = max(0, dot(vNormal, -lightDir));//Diffuse reflection (How much light is being reflected off of a surface)
	vec4 diffuse = d * lightColor;//Light Color
	vec4 base = texture(albedo, vUV);//Texture

	vertColor = vec4((diffuse * base).xyz, 1);//Combine it all together
}
